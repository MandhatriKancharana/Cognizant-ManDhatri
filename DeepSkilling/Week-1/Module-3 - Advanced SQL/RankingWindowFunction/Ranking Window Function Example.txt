Step 3: Create Table
CREATE TABLE Merchandise
(
    MerchandiseID INT PRIMARY KEY,
    MerchandiseName VARCHAR(50),
    SectionName VARCHAR(50),
    Amount DECIMAL(10,2)
);

Step 6: ROW_NUMBER()
SELECT
    MerchandiseID,
    MerchandiseName,
    SectionName,
    Amount,
    ROW_NUMBER() OVER
    (
        PARTITION BY SectionName
        ORDER BY Amount DESC
    ) AS RowNum
FROM Merchandise;

Step 7: RANK()
SELECT
    MerchandiseID,
    MerchandiseName,
    SectionName,
    Amount,
    RANK() OVER
    (
        PARTITION BY SectionName
        ORDER BY Amount DESC
    ) AS RankNum
FROM Merchandise;

Step 8: DENSE_RANK()
SELECT
    MerchandiseID,
    MerchandiseName,
    SectionName,
    Amount,
    DENSE_RANK() OVER
    (
        PARTITION BY SectionName
        ORDER BY Amount DESC
    ) AS DenseRankNum
FROM Merchandise;

Step 9: Final Assignment Query 
WITH RankedMerchandise AS
(
    SELECT
        MerchandiseID,
        MerchandiseName,
        SectionName,
        Amount,

        ROW_NUMBER() OVER
        (
            PARTITION BY SectionName
            ORDER BY Amount DESC
        ) AS RowNum,

        RANK() OVER
        (
            PARTITION BY SectionName
            ORDER BY Amount DESC
        ) AS RankNum,

        DENSE_RANK() OVER
        (
            PARTITION BY SectionName
            ORDER BY Amount DESC
        ) AS DenseRankNum

    FROM Merchandise
)

SELECT *
FROM RankedMerchandise
WHERE RowNum <= 3;
